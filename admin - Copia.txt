        function formatarData(dataISO) {
             if (typeof dataISO === 'string' && dataISO.includes('-')) {
                 return dataISO.split('-').reverse().join('/');
             }
             return String(dataISO);

        function formatarDataCompleta(dataISO) {
            const diasSemana = ["Dom", "Seg", "Ter", "Qua", "Qui", "Sex", "S√°b"];
            const data = new Date(dataISO + 'T00:00:00');
            const dia = diasSemana[data.getDay()];
            const dataFormatada = formatarData(dataISO);
            return `${dia}, ${dataFormatada}`;
        }

        function formatarPreco(valor) {
            if (valor === undefined || valor === null || isNaN(valor)) return 'R$ 0,00';
            return 'R$ ' + Number(valor).toFixed(2).replace('.', ',');
        }

        function getStartOfWeek(date) {
            const d = new Date(date);
            const day = d.getDay(); // 0 = Domingo, 1 = Segunda
            const diff = d.getDate() - day + (day === 0 ? -6 : 1); // Ajusta para Segunda-feira
            const start = new Date(d);
            start.setDate(diff);
            start.setHours(0, 0, 0, 0);
            return start;
        }

        async function adminAgendar() {
            const data = adminDataInput.value;
            const nome = document.getElementById('adminNomeCliente').value.trim();
            const servicoCompleto = document.getElementById('adminServicoSelecionado').value;
            const hora = document.getElementById('adminHorarioSelecionado').value;

                alert('Preencha data, nome, servi√ßo e hor√°rio.');
                return;
            }

            // Verifica data passada
            const selecionada = new Date(data + 'T00:00:00');
            const hoje = new Date();
            hoje.setHours(0, 0, 0, 0);

            if (selecionada.getTime() < hoje.getTime()) {
                alert('N√£o √© poss√≠vel agendar em datas passadas.');
                return;
            }
            
            // Determina a dura√ß√£o e pre√ßo
            let duracao = 40; 
            let preco = 50.00; // Pre√ßo Padr√£o

            if (servicoCompleto.includes("Combo")) { duracao = 80; preco = 90.00; }
            else if (servicoCompleto.includes("Barba")) { preco = 50.00; }
            else if (servicoCompleto.includes("Corte Infantil (Autismo)")) { duracao = 80; preco = 70.00; }
            else if (servicoCompleto.includes("Corte Infantil")) { preco = 40.00; }
            else if (servicoCompleto.includes("Corte")) { preco = 50.00; }

        async function renderAdminHorarios(ocupados = [], data) {
             const grid = document.getElementById('adminGridHorarios');
             grid.innerHTML = '';
             
             // Determina se o servi√ßo selecionado pelo admin exige 2 slots (80 min)
             const servicoSelecionado = document.getElementById('adminServicoSelecionado').value;
             const isDoisSlots = servicoSelecionado.includes("80");
             const hoje = new Date().toISOString().split('T')[0];

             horariosPossiveis.forEach(hora => {
                 const btn = document.createElement('button');
                 btn.className = 'time-btn';
                 btn.textContent = hora;
                 btn.value = hora;
                 
                 let deveBloquear = ocupados.includes(hora);
                 btn.title = '';

                 // Bloqueia hor√°rios passados
                 if (data === hoje) {
                    const agora = new Date();
                    const [h, m] = hora.split(':').map(Number);
                    const horaAgendamento = new Date(agora);
                    horaAgendamento.setHours(h, m, 0, 0);

                    // Adiciona um buffer de 10 minutos
                    const horaLimite = new Date(agora);
                    horaLimite.setMinutes(agora.getMinutes() + 10); 
                    
                    if (horaAgendamento.getTime() <= horaLimite.getTime()) {
                        deveBloquear = true;
                        btn.title = 'Hor√°rio j√° passou.';
                    }
                 }

                 if (!deveBloquear && isDoisSlots) {
                    const proximoHorario = getProximoHorario(hora);
                    if (!proximoHorario || ocupados.includes(proximoHorario)) {
                        deveBloquear = true;
                        btn.title = proximoHorario ? `O pr√≥ximo slot (${proximoHorario}h) est√° ocupado.` : `Sem slot seguinte para servi√ßo de 80 min.`;
                    } else {
                        btn.classList.add('combo-slot');
                    }
                 }

                 if (deveBloquear) {
                     btn.classList.add('blocked');
                     btn.disabled = true;
                     // desabilita option correspondente no select
                     if (selectHora) { 
                         const opt = Array.from(selectHora.options).find(o => o.value === hora); 
                         if (opt) opt.disabled = true; 
                     }
                 } else {
                     btn.classList.add('clickable');
                     btn.addEventListener('click', () => {
                         // remove sele√ß√£o anterior
                         document.querySelectorAll('#adminGridHorarios .time-btn').forEach(b => b.classList.remove('selected'));
                         btn.classList.add('selected');
                         adminHorarioSelecionadoVar = hora; 
                         // preenche select e foca no nome
                         if (selectHora) selectHora.value = hora;
                         const nomeEl = document.getElementById('adminNomeCliente');
                         if (nomeEl) nomeEl.focus();
                     });
                 }
                 grid.appendChild(btn);
             });
        }

        async function loadWeeklyAgenda(startOfWeekDate) {
            weeklyAppointmentsContainer.innerHTML = '<p class="empty-message" style="color:var(--primary);">Buscando agendamentos semanais...</p>';
            const startOfWeek = new Date(startOfWeekDate);
            startOfWeek.setHours(0, 0, 0, 0);
            const endOfWeek = new Date(startOfWeek);
            endOfWeek.setDate(startOfWeek.getDate() + 7); // 7 dias ap√≥s o in√≠cio (Segunda a Domingo)
            endOfWeek.setHours(23, 59, 59, 999);

            const startDateISO = startOfWeek.toISOString().split('T')[0];
            const endDateISO = endOfWeek.toISOString().split('T')[0];
            
            // Ajuste para exibir corretamente a data final
            const endOfWeekDisplay = new Date(startOfWeek);
            endOfWeekDisplay.setDate(startOfWeek.getDate() + 6);

            currentWeekDisplay.textContent = `${formatarData(startDateISO)} a ${formatarData(endOfWeekDisplay.toISOString().split('T')[0])}`;

            try {
                const q = query(
                    collection(db, "agendamentos"),
                    where("data", ">=", startDateISO),
                    where("data", "<", endDateISO),
                    orderBy("data", "asc"),
                    orderBy("hora", "asc")
                );
                const snapshot = await getDocs(q);

                if (snapshot.empty) {
                    weeklyAppointmentsContainer.innerHTML = '<p class="empty-message">Nenhum agendamento encontrado para esta semana.</p>';
                    return;
                }

                const agendamentosPorDia = {};
                snapshot.forEach(docSnapshot => {
                    const agendamento = docSnapshot.data();
                    const data = agendamento.data;
                    if (!agendamentosPorDia[data]) {
                        agendamentosPorDia[data] = [];
                    }
                    agendamentosPorDia[data].push({ id: docSnapshot.id, ...agendamento });
                });

                weeklyAppointmentsContainer.innerHTML = '';
                
                let currentDate = new Date(startOfWeek);
                // Renderiza os 7 dias da semana
                for (let i = 0; i < 7; i++) {
                    const dateISO = currentDate.toISOString().split('T')[0];
                    const dayAppointments = agendamentosPorDia[dateISO] || [];
                    
                    const dayContainer = document.createElement('div');
                    dayContainer.classList.add('weekly-day-container');
                    
                    let dayContent = '';
                    
                    if (dayAppointments.some(a => a.idBloqueio === "BLOQUEIO_TOTAL")) {
                        dayContent = '<p class="agendamento-item dia-fechado">‚ùå DIA FECHADO</p>';
                    } else if (dayAppointments.length === 0) {
                        dayContent = '<p style="margin-top: 10px; color: #888; font-size: 0.9em; text-align: center;">Nenhum agendamento.</p>';
                    } else {
                        dayAppointments.forEach(agendamento => {
                            const isSlotExtra = agendamento.parentId || (agendamento.servico && agendamento.servico.toString().toUpperCase().includes('SLOT_EXTRA'));
                            const isCortesia = agendamento.preco === 0 && agendamento.observacoes && agendamento.observacoes.includes('Cortesia'); 
                            const servicoCurto = agendamento.servico.split('(')[0].trim();
                            const precoDisplay = isCortesia ? 'CORTESIA' : formatarPreco(agendamento.preco);
                            const corBorda = isCortesia ? 'var(--red)' : (isSlotExtra ? '#585800' : 'var(--primary)');
                            
                            dayContent += `
                                <div class="agendamento-item ${isSlotExtra ? 'slot-extra' : ''}" onclick="openModal('${agendamento.id}')" style="margin-bottom: 5px; border-left-color: ${corBorda};">
                                    <p style="margin: 0;">
                                        <span style="font-weight: bold;">${agendamento.hora}h</span> | 
                                        ${agendamento.nome} - 
                                        ${servicoCurto} (${precoDisplay})
                                    </p>
                                </div>
                            `;
                        });
                    }
                    
                    dayContainer.innerHTML = `
                        <h4>${formatarDataCompleta(dateISO)}</h4>
                        ${dayContent}
                    `;
                    weeklyAppointmentsContainer.appendChild(dayContainer);
                    
                    currentDate.setDate(currentDate.getDate() + 1);
                }

            } catch (error) {
                console.error("Erro ao carregar agenda semanal:", error);
                weeklyAppointmentsContainer.innerHTML = `<p class="error-message">Erro ao carregar agenda semanal.</p>`;
            }
        }

        function normalizarServico(servico) { 
             if (!servico) return 'Servi√ßo Indefinido'; 
             const svc = String(servico).trim(); 
             
             if (svc.includes('Combo') || svc.includes('Combo Cabelo + Barba')) { return 'Combo Cabelo e Barba'; } 
             if (svc.includes('Corte Infantil (Autismo)')) { return 'Corte Infantil - Autismo'; } 
             if (svc.includes('Corte Infantil')) { return 'Corte Infantil'; } 
             if (svc.includes('Corte') && !svc.includes('Infantil')) { return 'Corte Simples'; } 
             if (svc.includes('Barba')) { return 'Barba'; } 
             
             // Remove dura√ß√£o (Ex: - 40 min)
             return svc.split(' - ')[0].trim();
        }

        async function generateReport(period) { 
            adminReportMsg.style.display = 'block';
            adminReportMsg.textContent = `Buscando dados de vendas (${period})...`; 
            
            const start = new Date();
            let periodDisplay = '';

            // L√≥gica para definir o intervalo de datas
            if (period === 'weekly') {
                periodDisplay = '√öltimos 7 Dias';
                start.setDate(start.getDate() - 7);
            } else if (period === 'monthly') {
                periodDisplay = '√öltimos 30 Dias';
                start.setDate(start.getDate() - 30);
            } else if (period === 'annual') {
                periodDisplay = '√öltimos 365 Dias';
                start.setDate(start.getDate() - 365);
            }
            start.setHours(0, 0, 0, 0);
            const startISO = isoDateString(start);
            const end = new Date(); 
            end.setHours(23, 59, 59, 999);
            
            const agendamentosRef = collection(db, "agendamentos");
            const q = query(agendamentosRef, where("data", ">=", startISO), orderBy("data", "asc"), orderBy("hora", "asc"));
            
            // Checa se o usu√°rio atual √© o admin especial (voc√™)
            const isAdminSpecial = currentUser && currentUser.email.toLowerCase() === SAULLINHO_SPECIAL_ADMIN; 

            try {
                const snap = await getDocs(q);
                
                let totalRevenue = 0; 
                const serviceRevenue = {}; 
                const mains = []; 
                let totalCortes = 0; // Contador de cortes/combos para a taxa
                let totalTax = 0; // Total da taxa do desenvolvedor

                snap.forEach(docSnap => { 
                    const d = docSnap.data(); 
                    // pula documentos de bloqueio de dia e slots extras
                    if (d && (d.idBloqueio === 'BLOQUEIO_TOTAL' || d.parentId || (d.servico && d.servico.toString().toUpperCase().includes('SLOT_EXTRA')))) return;
                    
                    const preco = Number(d.preco) || 0;
                    const servicoNome = normalizarServico(d.servico);

                    // Conta cortes/combos para a taxa (s√≥ cortes e combos pagam taxa)
                    const isCorteOuCombo = servicoNome.includes('Corte') || servicoNome.includes('Combo');
                    
                    if (isCorteOuCombo) {
                        totalCortes++;
                        if (isAdminSpecial) {
                            totalTax += CORTE_TAX;
                        }
                    }

                    if (preco > 0) {
                        totalRevenue += preco;
                        serviceRevenue[servicoNome] = (serviceRevenue[servicoNome] || 0) + preco;
                    }
                    
                    mains.push({ id: docSnap.id, data: d }); 
                }); 

                // Transforma o mapa de receitas em lista para o renderer
                const serviceList = Object.entries(serviceRevenue).map(([servico, total]) => ({ servico, total }));
                
                if (mains.length === 0) { 
                    adminReportMsg.style.color = 'var(--red)'; 
                    adminReportMsg.textContent = `Nenhum agendamento principal com receita encontrado para o per√≠odo.`; 
                    reportResults.innerHTML = '<p class="empty-message">Nenhuma receita encontrada para o per√≠odo.</p>';
                    return; 
                } 

                // 1. Renderiza o relat√≥rio na tela 
                const taxData = {
                    totalCortes: totalCortes,
                    totalTax: totalTax
                };
                renderServiceReports(serviceList, totalRevenue, periodDisplay, isAdminSpecial, taxData); 
                
                adminReportMsg.style.color = 'var(--green)';
                adminReportMsg.textContent = `Relat√≥rio de ${periodDisplay} gerado com sucesso! ${totalCortes} cortes/combos encontrados.`;

            } catch (e) { 
                console.error('Erro gerando relat√≥rio:', e); 
                adminReportMsg.style.color = 'var(--red)'; 
                adminReportMsg.textContent = 'Erro ao gerar relat√≥rio. Veja o console.'; 
            } 
        }

        async function loadExpenses(period = 'currentMonth') {
            expenseList.innerHTML = '<p class="empty-message" style="color:var(--red);">Carregando despesas...</p>';

            let startDateISO;
            const today = new Date();
            today.setHours(0, 0, 0, 0);

            let start = new Date(today);
            
            // L√≥gica para definir o intervalo de datas (para a query do Firebase: date >= startDateISO)
            if (period === 'currentMonth') {
                start.setDate(1); // Primeiro dia do m√™s
                startDateISO = start.toISOString().split('T')[0];
            } else if (period === 'last30Days') {
                start.setDate(today.getDate() - 30);
                startDateISO = start.toISOString().split('T')[0];
            } else if (period === 'currentYear') {
                start.setMonth(0); 
                start.setDate(1); 
                startDateISO = start.toISOString().split('T')[0];
            } else if (period === 'all') {
                startDateISO = '1900-01-01'; // Busca tudo
            }
            
            let useInMemoryFilter = (period === 'lastMonth');
            let lastMonthStart = '';
            let lastMonthEnd = '';
            
            if(useInMemoryFilter) {
                // Filtro para M√™s Passado:
                let tempStart = new Date(today);
                tempStart.setMonth(today.getMonth() - 1);
                tempStart.setDate(1); 
                lastMonthStart = tempStart.toISOString().split('T')[0];

                let tempEnd = new Date(today);
                tempEnd.setDate(0); // √öltimo dia do m√™s passado
                lastMonthEnd = tempEnd.toISOString().split('T')[0];

                // Para garantir que a query do Firebase pegue os dados, for√ßamos o in√≠cio da query para o in√≠cio do m√™s passado.
                startDateISO = lastMonthStart; 
            }

            try {
                // A query precisa ter orderBy("date") para que where("date", ">=") funcione
                const q = query(
                    collection(db, "despesas"),
                    where("date", ">=", startDateISO),
                    orderBy("date", "desc"), // Ordena da mais recente para a mais antiga
                );
                const snapshot = await getDocs(q);

                let total = 0;
                let expenses = [];
                let itemsHTML = '';

                snapshot.forEach(docSnap => {
                    const d = docSnap.data();
                    
                    // Filtro em mem√≥ria para 'lastMonth' (garante que n√£o pega o m√™s atual)
                    if (useInMemoryFilter) {
                        if (d.date < lastMonthStart || d.date > lastMonthEnd) return;
                    }

                    expenses.push({ id: docSnap.id, ...d });
                    total += Number(d.value) || 0;
                });

                if (expenses.length === 0) {
                    expenseList.innerHTML = `<p class="empty-message">Nenhuma despesa encontrada para o per√≠odo selecionado.</p>`;
                } else {
                    expenses.forEach(d => {
                        itemsHTML += `
                            <div class="agendamento-item" style="border-left-color: var(--red);">
                                <div class="agendamento-info">
                                    <p style="font-weight: bold;">
                                        üî¥ ${d.description} 
                                    </p>
                                    <p>
                                        üí∏ ${formatarPreco(d.value)} | üè∑Ô∏è ${d.category}
                                    </p>
                                    <p style="font-size: 0.9em; color: #ccc;">
                                        üìÖ Data da Despesa: ${formatarData(d.date)}
                                    </p>
                                </div>
                                <div class="agendamento-actions">
                                    <button class="cancel-admin-btn" style="background: var(--red); padding: 5px 10px;" onclick="cancelarDespesa('${d.id}', '${d.description}', ${d.value})"> 
                                        Excluir 
                                    </button>
                                </div>
                            </div>
                        `;
                    });
                    expenseList.innerHTML = itemsHTML;
                }

                // Atualiza o total (Com a cor vermelha de despesa)
                expenseTotalDisplay.innerHTML = `
                    <h3 style="margin: 0 0 5px 0;">üî¥ TOTAL DE DESPESAS NO PER√çODO</h3>
                    <div style="font-size: 1.5em; font-weight: bold;">${formatarPreco(total)}</div>
                `;

            } catch (e) {
                console.error('Erro ao carregar despesas:', e);
                expenseList.innerHTML = `<p class="error-message">Erro ao carregar despesas. Verifique a conex√£o e crie o index 'date' no Firebase se necess√°rio (a mensagem de erro do console geralmente informa o link para criar).</p>`;
            }
        }

        // Listener para o filtro
        expenseFilterSelect.addEventListener('change', (e) => loadExpenses(e.target.value));

        // Fun√ß√£o global para excluir despesa
        window.cancelarDespesa = async function(id, description, value) {
            const confirmacao = confirm(`Tem certeza que deseja EXCLUIR a despesa "${description}" no valor de ${formatarPreco(value)}?`);
            if (!confirmacao) return;
            try {
                await deleteDoc(doc(db, "despesas", id));
                alert(`Despesa exclu√≠da com sucesso.`);
                loadExpenses(expenseFilterSelect.value); // Recarrega a lista
            } catch (e) {
                console.error("Erro ao excluir despesa:", e);
                alert("Erro ao excluir despesa. Verifique as regras de seguran√ßa.");
            }
        }
